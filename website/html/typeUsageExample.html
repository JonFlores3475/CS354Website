TypeScript Examples

1. Easy Example: Usage of Interfaces and Types

Code:


// Define an interface for a User
interface User {
  id: number;
  name: string;
  email: string;
}

// Define a type for a list of users
type UserList = User[];

// Create a variable using the User interface
const user1: User = {
  id: 1,
  name: "Alice",
  email: "alice@example.com",
};

// Create a list of users using the UserList type
const users: UserList = [
  user1,
  { id: 2, name: "Bob", email: "bob@example.com" },
  { id: 3, name: "Charlie", email: "charlie@example.com" },
];

// Function to display user info
function displayUsers(users: UserList): void {
  users.forEach((user) => {
    console.log(`ID: ${user.id}, Name: ${user.name}, Email: ${user.email}`);
  });
}

// Call the function
displayUsers(users);

Expected Output in Terminal:


ID: 1, Name: Alice, Email: alice@example.com
ID: 2, Name: Bob, Email: bob@example.com
ID: 3, Name: Charlie, Email: charlie@example.com

2. Medium Example: Usage of Storage Containers with Generic Type <T>

Code:


// Generic storage container
class StorageContainer {
  private items: T[] = [];

  // Add an item to the storage
  add(item: T): void {
    this.items.push(item);
  }

  // Get all items
  getAll(): T[] {
    return this.items;
  }

  // Get an item by index
  get(index: number): T | undefined {
    return this.items[index];
  }
}

// Create a storage container for numbers
const numberStorage = new StorageContainer();
numberStorage.add(1);
numberStorage.add(2);
numberStorage.add(3);
console.log("Number Storage:", numberStorage.getAll());

// Create a storage container for strings
const stringStorage = new StorageContainer();
stringStorage.add("TypeScript");
stringStorage.add("Generics");
console.log("String Storage:", stringStorage.getAll());

// Example of using get
console.log("First String:", stringStorage.get(0));

Expected Output in Terminal:


Number Storage: [ 1, 2, 3 ]
String Storage: [ 'TypeScript', 'Generics' ]
First String: TypeScript
